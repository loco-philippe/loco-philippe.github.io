 61        '''return default coupling codec between two keys list and optionaly if
 62        the relationship is distributed'''
 63        if not key1 or not key2:
 64            return 0
 65        k1k2 = [tuple((v1, v2)) for v1, v2 in zip(key1, key2)]
 66        dist = len(Cutil.tocodec(k1k2))
 67        if not distr:
 68            return dist
 69        distrib = False
 70        if dist == (max(key1) + 1) * (max(key2) + 1):
 71            distrib = max(Counter(k1k2).values()) == len(key1) // dist
 72        return [dist, distrib]

 74    @staticmethod
 75    def encode_coef(lis):
 76        '''Generate a repetition coefficient for periodic list'''
 77        if len(lis) < 2:
 78            return 0
 79        coef = 1
 80        while coef != len(lis):
 81            if lis[coef-1] != lis[coef]:
 82                break
 83            coef += 1
 84        if (not len(lis) % (coef * (max(lis) + 1)) and
 85                lis == Cutil.keysfromcoef(coef, max(lis) + 1, len(lis))):
 86            return coef
 87        return 0

 91        '''return the function func applied to the object value with parameters args and kwargs'''
 92        if func in (None, []):
 93            return value
 94        lis = []
 95        if not (isinstance(value, list) or value.__class__.__name__ in ['Cfield', 'Cdataset']):
 96            listval = [value]
 97        else:
 98            listval = value
 99        for val in listval:
100            try:
101                lis.append(val.func(*args, **kwargs))
102            except:
103                try:
104                    lis.append(func(val, *args, **kwargs))
105                except:
106                    try:
107                        lis.append(listval.func(val, *args, **kwargs))
108                    except:
109                        try:
110                            lis.append(func(listval, val, *args, **kwargs))
111                        except:
112                            raise FieldError("unable to apply func")
113        if len(lis) == 1:
114            return lis[0]

441    def from_ntv(cls, ntv_value=None, extkeys=None, reindex=True, decode_str=False,
442                 add_type=True, lengkeys=None):
443        '''Generate an Field Object from a Ntv field object'''
444        if isinstance(ntv_value, cls):
445            return copy(ntv_value)
446        if ntv_value is None:
447            return cls()
448        ntv = Ntv.obj(ntv_value, decode_str=decode_str)
449        #ntv = NtvList(ntv_value)
450        name, typ, codec, parent, keys, coef, leng = NtvUtil.decode_ntv_tab(
451            ntv, cls.ntv_to_val)
452        if parent and not extkeys:
453            return None
454        if coef:
455            keys = Cutil.keysfromcoef(coef, leng//coef, lengkeys)
456        elif extkeys and parent:
457            keys = Cutil.keysfromderkeys(extkeys, keys)
458        elif extkeys and not parent:
459            keys = extkeys
460        keys = list(range(len(codec))) if keys is None else keys
461        name = ntv.json_name(string=True) if add_type else name
462        return cls(codec=codec, name=name, keys=keys, reindex=reindex)

562        '''
563        Transform indexes in coupled or derived indexes (codec extension).
564        If derived option is True, self._codec is extended and idx codec not,
565        else, both are coupled and both codec are extended.
566
567        *Parameters*
568
569        - **idx** : single Field or list of Field to be coupled or derived.

573        (only for self index)

575        with default codec. But if not derived, idx indexes MUST to be reindexed.

578        if not isinstance(idx, list):
579            index = [idx]
580        else:
581            index = idx
582        idxzip = self.__class__(list(zip(*([self.keys] + [ix.keys for ix in index]))),
583                                reindex=True)
584        self.tocoupled(idxzip)
585        if not derived:
586            for ind in index:
587                ind.tocoupled(idxzip)
588        if duplicate:
589            return self.getduplicates(reindex)
590        if reindex:
591            self.reindex()
592        return None
