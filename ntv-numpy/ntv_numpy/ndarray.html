221    @staticmethod
222    def read_json(jsn, **kwargs):
223        ''' convert json ntv_value into a ndarray.
224
225        *Parameters*
226
227        - **convert** : boolean (default True) - If True, convert json data with
228        non Numpy ntv_type into data with python type
229        '''
230        option = {'convert': True} | kwargs
231        jso = json.loads(jsn) if isinstance(jsn, str) else jsn
232        ntv_value, = Ntv.decode_json(jso)[:1]
233
234        ntv_type = None
235        shape = None
236        match ntv_value[:-1]:
237            case []: ...
238            case [ntv_type, shape]: ...
239            case [str(ntv_type)]: ...
240            case [list(shape)]: ...
241        unidim = not shape is None
242        if isinstance(ntv_value[-1], str):
243            return Ndarray(ntv_value[-1], shape=shape, ntv_type=ntv_type)
244        darray = Darray.read_json(ntv_value[-1], dtype=NpUtil.dtype(ntv_type), 
245                                  unidim=unidim)
246        darray.data = NpUtil.convert(ntv_type, darray.data, tojson=False,
247                                     convert=option['convert'])
248        return Ndarray(darray.values, shape=shape, ntv_type=ntv_type)

250    def to_json(self, **kwargs):
251        ''' convert a Ndarray into json-value
252
253        *Parameters*
254
255        - **noshape** : Boolean (default True) - if True, without shape if dim < 1
256        - **notype** : Boolean (default False) - including data type if False
257        - **novalue** : Boolean (default False) - including value if False
258        - **format** : string (default 'full') - representation format of the ndarray,
259        - **encoded** : Boolean (default False) - json-value if False else json-text
260        - **header** : Boolean (default True) - including ndarray type
261        '''
262        option = {'format': 'full', 'header': True, 'encoded': False,
263                  'notype': False, 'noshape': True, 'novalue': False} | kwargs
264        if self.mode in ['undefined', 'inconsistent']:
265            return None
266        if self.mode == 'absolute' and len(self.darray) == 0:
267            return [[]]
268
269        shape = None if not self.shape or (len(self.shape) < 2 and 
270                                           option['noshape']) else self.shape
271
272        if self.mode == 'relative':
273            js_val = self.uri
274        else:
275            js_val = NpUtil.ntv_val(self.ntv_type, self.darray, option['format'],
276                                    self.is_json) if not option['novalue'] else ['-']
277
278        lis = [self.ntv_type if not option['notype'] else None, shape, js_val]
279        return NpUtil.json_ntv(None, 'ndarray',
280                               [val for val in lis if not val is None],
281                               header=option['header'], encoded=option['encoded'])

283    @staticmethod
284    def equals(nself, nother):
285        '''return True if all elements are equals and dtype are equal'''
286        if not (isinstance(nself, np.ndarray) and isinstance(nother, np.ndarray)):
287            return False
288        if nself.dtype != nother.dtype or nself.shape != nother.shape:
289            return False
290        if len(nself.shape) == 0:
291            return True
292        if len(nself) != len(nother):
293            return False
294        if len(nself) == 0:
295            return True
296        if isinstance(nself[0], (np.ndarray, pd.Series, pd.DataFrame)):
297            SeriesConnec = NtvConnector.connector().get('SeriesConnec')
298            DataFrameConnec = NtvConnector.connector().get('DataFrameConnec')
299            equal = {np.ndarray: Ndarray.equals,
300                     pd.Series: SeriesConnec.equals,
301                     pd.DataFrame: DataFrameConnec.equals}
302            for nps, npo in zip(nself, nother):
303                if not equal[type(nself[0])](nps, npo):
304                    return False
305            return True
306        return np.array_equal(nself, nother)

392    @staticmethod
393    def is_json(obj):
394        ''' check if obj is a json structure and return True if obj is a json-value
395
396        *Parameters*
397
398        - **obj** : object to check'''
399        if obj is None:
400            return True
401        is_js = NtvConnector.is_json
402        match obj:
403            case str() | int() | float() | bool():
404                return True
405            case list() | tuple() as obj:
406                if not obj:
407                    return True
408                return min(is_js(obj_in) for obj_in in obj)
409            case dict() as obj:
410                if not obj:
411                    return True
412                if not min(isinstance(key, str) for key in obj.keys()):
413                    return False
414                return min(is_js(obj_in) for obj_in in obj.values())
415            case _:
416                return False

418    @staticmethod
419    def convert(ntv_type, nda, tojson=True, convert=True):
420        ''' convert ndarray with external NTVtype.
421
422        *Parameters*
423
424        - **ntv_type** : string - NTVtype deduced from the ndarray name_type and dtype,
425        - **nda** : ndarray to be converted.
426        - **tojson** : boolean (default True) - apply to json function
427        - **convert** : boolean (default True) - If True, convert json data with
428        non Numpy ntv_type into data with python type
429        '''
430        if tojson:
431            match ntv_type:
432                case dat if dat in NpUtil.DATATION_DT:
433                    return nda.astype(NpUtil.DATATION_DT[dat]).astype(str)
434                case 'bytes':
435                    return nda.astype('bytes').astype(str)
436                case 'time':
437                    return nda.astype(str)
438                case 'decimal64':
439                    return nda.astype(float)
440                case 'geojson':
441                    return np.frompyfunc(ShapelyConnec.to_geojson, 1, 1)(nda)
442                case _:
443                    return nda
444        else:
445            match [ntv_type, convert]:
446                case [None, _]:
447                    return nda
448                case [dat, _] if dat in NpUtil.DATATION_DT:
449                    return nda.astype(NpUtil.DATATION_DT[dat])
450                case [std, _] if std in NpUtil.OTHER_DT:
451                    return nda.astype(NpUtil.OTHER_DT[std])
452                case ['time', True]:
453                    return np.frompyfunc(datetime.time.fromisoformat, 1, 1)(nda)
454                case ['decimal64', True]:
455                    return np.frompyfunc(Decimal, 1, 1)(nda)
456                case ['narray', True]:
457                    nar = np.frompyfunc(Ndarray.read_json, 1, 1)(nda)
458                    return np.frompyfunc(Ndarray.to_ndarray, 1, 1)(nar)
459                case ['ndarray', True]:
460                    return np.frompyfunc(Ndarray.read_json, 1, 1)(nda)
461                case [python, _] if python in NpUtil.PYTHON_DT:
462                    return nda.astype('object')
463                case [connec, True] if connec in NpUtil.CONNECTOR_DT:
464                    return np.fromiter([NtvConnector.uncast(nd, None, connec)[0]
465                                        for nd in nda], dtype='object')
466                case [('point' | 'line' | 'polygon' | 'geometry'), True]:
467                    return np.frompyfunc(ShapelyConnec.to_geometry, 1, 1)(nda)
468                case [_, False]:
469                    return nda.astype(NpUtil.CONVERT_DT[
470                        NpUtil.dtype(ntv_type, convert=False)])
471                case _:
472                    return nda.astype(NpUtil.dtype(ntv_type))

474        # float.fromhex(x.hex()) == x, bytes(bytearray.fromhex(x.hex())) == x 
475    @staticmethod
476    def ntv_val(ntv_type, nda, form, is_json=False):
477        ''' convert a np.ndarray into NTV json-value.
478
479        *Parameters*
480
481        - **ntv_type** : string - NTVtype deduced from the ndarray, name_type and dtype,
482        - **nda** : ndarray to be converted.
483        - **form** : format of data ('full', 'complete', 'sparse', 'primary').
484        - **is_json** : boolean (defaut False) - True if nda data is Json data
485        '''
486        if form == 'complete' and len(nda) < 2:
487            raise NdarrayError(
488                "complete format is not available with ndarray length < 2")
489        Format = NpUtil.FORMAT_CLS[form]
490        darray = Format(nda)
491        ref = darray.ref
492        coding = darray.coding
493        if is_json:
494            return Format(darray.data, ref=ref, coding=coding).to_json()
495        match ntv_type:
496            case 'narray':
497                data = [Ndarray(nd).to_json(header=False) for nd in darray.data]
498            case 'ndarray':
499                #data = [Ndarray.to_json(nd) for nd in darray.data]
500                data = [Ndarray(nd).to_json(header=False) for nd in darray.data]
501            case connec if connec in NpUtil.CONNECTOR_DT:
502                data = [NtvConnector.cast(nd, None, connec)[0]
503                        for nd in darray.data]
504            case 'point' | 'line' | 'polygon' | 'geometry':
505                data = np.frompyfunc(ShapelyConnec.to_coord, 1, 1)(darray.data)
506            case None:
507                data = nda
508            case _:
509                data = NpUtil.convert(ntv_type, darray.data)
510        return Format(data, ref=ref, coding=coding).to_json()
