132    def __init__(self, codec=None, name=None, keys=None,
133                 lendefault=0, reindex=False, fast=False):
134        '''
135        Field constructor.
136
137        *Parameters*
138
139        - **codec** :  list (default None) - external different values of index (see data model)
140        - **keys** :  list (default None)  - key value of index (see data model)
141        - **name** : string (default None) - name of index (see data model)
142        - **lendefault** : integer (default 0) - default len if no keys is defined
143        - **reindex** : boolean (default True) - if True, default codec is apply
144        - **fast**: boolean (default False) - if True, codec is created without conversion'''
145        if isinstance(codec, Field):
146            Cfield.__init__(self, deepcopy(codec._codec), copy(codec.name), copy(codec._keys))
147            return
148        if codec is None:
149            codec = []
150        if not isinstance(codec, list):
151            codec = [codec]
152        leng = lendefault
153        if codec and len(codec) > 0 and not leng:
154            leng = len(codec)
155        if not keys is None:
156            leng = len(keys)
157        if not name:
158            name = DEFAULTINDEX
159        if not (keys is None or isinstance(keys, list)):
160            raise FieldError("keys not list")
161        if keys is None and leng == 0:
162            keys = []
163        elif keys is None:
164            keys = [(i*len(codec))//leng for i in range(leng)]
165        if codec == []:
166            keysset = Cutil.tocodec(keys)
167            codec = self.l_to_i(keysset, fast=True)
168        codec = self.l_to_i(codec, fast=fast)
169        Cfield.__init__(self, codec, name, keys, reindex=reindex)

172        ''' modify values item'''

174            raise FieldError("out of bounds")

179        '''Create a new Field with values are tuples of listidx Field values
