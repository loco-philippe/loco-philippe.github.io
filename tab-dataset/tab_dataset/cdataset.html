161    def from_ntv(cls, ntv_value, reindex=True, decode_str=False, fast=False):
162        '''Generate an Dataset Object from a ntv_value
163
164        *Parameters*
165
166        - **ntv_value** : bytes, string, Ntv object to convert
167        - **reindex** : boolean (default True) - if True, default codec for each Field
168        - **decode_str**: boolean (default False) - if True, string are loaded in json data
169        - **fast** : boolean (default False) - if True, ntv_value are not converted in json-value'''
170        ntv = Ntv.obj(ntv_value, decode_str=decode_str, fast=fast)
171        if len(ntv) == 0:
172            return cls()
173        lidx = [list(NtvUtil.decode_ntv_tab(
174            ntvf, cls.field_class.ntv_to_val)) for ntvf in ntv]
175        leng = max([idx[6] for idx in lidx])
176        for ind in range(len(lidx)):
177            if lidx[ind][0] == '':
178                lidx[ind][0] = 'i'+str(ind)
179            NtvConnector.init_ntv_keys(ind, lidx, leng)
180        lindex = [cls.field_class(idx[2], idx[0], idx[4], None,  # idx[1] pour le type,
181                                  reindex=reindex) for idx in lidx]
182        return cls(lindex, reindex=reindex, name=ntv.name)

185        ''' Add other's values to self's values for each index

188
189        - **other** : Dataset object to add to self object
190        - **name** : Boolean (default False) - Add values with same index name (True) or

199            raise DatasetError('name are not identical')
200        for i in range(self.lenindex):

277        Change the order of the index .

283        *Returns* : self '''

292    def check_relation(self, field, parent, typecoupl, value=True):
293        f_parent = self.nindex(parent) if isinstance(
294            parent, str) else self.lindex[parent]
295        f_field = self.nindex(field) if isinstance(
296            field, str) else self.lindex[field]
297        match typecoupl:
298            case 'derived':
299                errors = f_parent.coupling(f_field, reindex=True)
300            case 'coupled':
301                errors = f_parent.coupling(
302                    f_field, derived=False, reindex=True)
303            case _:
304                raise DatasetError(typecoupl + "is not a valid relationship")
305        if not value:
306            return errors
307        return {'row': list(errors), f_field.name: f_field[errors], f_parent.name: f_parent[errors]}

309    def check_relationship(self, relations):
310        '''get the inconsistent records for each relationship defined in relations
311
312         *Parameters*
313
314        - **relations** : list of dict or single dict - list of fields with relationship property
315
316        *Returns* :
317
318        - dict with for each relationship: key = string with the two fields name,
319        and value = list of inconsistent records
320        - or if single relationship : value'''
321        if not isinstance(relations, (list, dict)):
322            raise DatasetError("relations is not correct")
323        if isinstance(relations, dict):
324            relations = [relations]
325        dic_res = {}
326        for field in relations:
327            if not 'relationship' in field or not 'name' in field:
328                continue
329            if not 'parent' in field['relationship'] or not 'link' in field['relationship']:
330                raise DatasetError("relationship is not correct")
331            rel = field['relationship']['link']
332            f_parent = field['relationship']['parent']
333            f_field = field['name']
334            name_rel = f_field + ' - ' + f_parent
335            if self.nindex(f_parent) is None or self.nindex(f_field) is None:
336                raise DatasetError("field's name is not present in data")
337            dic_res[name_rel] = self.check_relation(
338                f_field, f_parent, rel, False)
339        if len(dic_res) == 1:
340            return list(dic_res.values())[0]
341        return dic_res
