123    def __init__(self, codec=None, name=None, keys=None, typevalue=ES.def_clsName,
124                 lendefault=0, reindex=False, castobj=True):
125        '''
126        Iindex constructor.
127
128        *Parameters*
129
130        - **codec** :  list (default None) - external different values of index (see data model)
131        - **keys** :  list (default None)  - key value of index (see data model)
132        - **name** : string (default None) - name of index (see data model)
133        - **typevalue** : string (default ES.def_clsName) - typevalue to apply to codec
134        - **lendefault** : integer (default 0) - default len if no keys is defined
135        - **reindex** : boolean (default True) - if True, default codec is apply'''
136        if isinstance(codec, Iindex):
137            self._keys = copy(codec._keys)
138            self._codec = deepcopy(codec._codec)
139            self.name = copy(codec.name)
140            return
141        if codec is None:
142            codec = []
143        if not isinstance(codec, list):
144            codec = [codec]
145        leng = lendefault
146        if codec and len(codec) > 0 and not leng:
147            leng = len(codec)
148        if not keys is None:
149            leng = len(keys)
150        if not name:
151            name = ES.defaultindex
152        typevalue = util.typename(name, typevalue)
153        if not (keys is None or isinstance(keys, list)):
154            raise IindexError("keys not list")
155        if keys is None and leng == 0:
156            keys = []
157        elif keys is None:
158            keys = [(i*len(codec))//leng for i in range(leng)]
159        if not isinstance(codec, list):
160            raise IindexError("codec not list")
161        if codec == []:
162            codec = util.tocodec(keys)
163        if not typevalue is None or castobj:
164            codec = util.castobj(codec, typevalue)
165        self._keys = keys
166        self._codec = codec
167        self.name = name
168        if reindex:
169            self.reindex()

281    @classmethod
282    def from_dict_obj(cls, bsd, typevalue=ES.def_clsName, reindex=False):
283        '''Generate an Iindex Object from a dict value'''
284        var = False
285        if not isinstance(bsd, dict):
286            raise IindexError('data is not a dict')
287        name = list(bsd.keys())[0]
288        bsdv = list(bsd.values())[0]
289        if not 'value' in bsdv:
290            raise IindexError('value is not present')
291        value = bsdv['value']
292        if not isinstance(value, list):
293            value = [value]
294        if 'type' in bsdv and isinstance(bsdv['type'], str):
295            typevalue = bsdv['type']
296        if 'var' in bsdv and isinstance(bsdv['var'], bool):
297            var = bsdv['var']
298        codec = [util.castval(val['codec'], typevalue) for val in value]
299        pairs = []
300        for i, rec in enumerate(value):
301            record = rec['record']
302            if not isinstance(record, list):
303                record = [record]
304            for j in record:
305                pairs.append((j, i))
306        if not pairs:
307            return (var, cls())
308        keys = list(list(zip(*sorted(pairs)))[1])
309
310        idx = cls(name=name, codec=codec, keys=keys, typevalue=None)
311        return (var, idx)

411    def add(self, other, solve=True):
412        ''' Add other's values to self's values
413
414        *Parameters*
415
416        - **other** : Iindex object to add to self object
417        - **solve** : Boolean (default True) - If True, replace None other's codec value
418        with self codec value.
419
420        *Returns* : self '''
421        if solve:
422            solved = copy(other)
423            for i in range(len(solved._codec)):
424                if not util.isNotNull(solved._codec[i]) and i in range(len(self._codec)):
425                    solved._codec[i] = self._codec[i]
426            values = self.values + solved.values
427        else:
428            values = self.values + other.values
429        codec = util.tocodec(values)
430        if set(codec) != set(self._codec):
431            self._codec = codec
432        self._keys = util.tokeys(values, self._codec)
433        return self

452        '''return dict with lencodec, typecodec, ratecodec, mincodec, maxcodec'''
453        maxi = len(self)
454        mini = len(set(self._codec))
455        xlen = len(self._codec)
456        rate = 0.0
457        if maxi == 0:
458            typecodec = 'null'
459        elif xlen == 1:
460            typecodec = 'unique'
461        elif mini == maxi:
462            typecodec = 'complete'
463        elif xlen == maxi:
464            typecodec = 'full'
465        else:
466            rate = (maxi - xlen) / (maxi - mini)
467            if xlen == mini:
468                typecodec = 'default'
469            else:
470                typecodec = 'mixed'
471        return {'lencodec': xlen, 'mincodec': mini, 'maxcodec': maxi,
472                'typecodec': typecodec, 'ratecodec': rate}
