Created on Thu Oct 12 14:49:34 2023

@author: philippe@loco-labs.io

This module analyses structure and relationships included in a tabular object (Pandas DataFrame, Dataset, list of list) :

  • Structure of a single field (class AnaField),
  • Relationship between two fields (class AnaRelation)
  • Structure and relationships of fields inside a dataset (class AnaDfield)
  • Structure of a dataset (class AnaDataset)

It contains the classes analysis.AnaField, analysis.AnaRelation, analysis.AnaDfield, analysis.AnaDataset, analysis.Util, analysis.AnaError.

503            return ROOTED

511    @property 
512    def p_derived(self):
513        '''return the derived parent of the AnaDfield'''
514        if self.category in (UNIQUE, ROOTED, ROOTDERIVED):
515            return self.dataset.root
516        if self.category == COUPLED:
517            return [rel.relation[1] for rel in self.list_coupled 
518                    if not rel.relation[1].category == COUPLED][0]
519        distance_min = min([rel.distance for rel in self.list_p_derived])
520        for rel in self.list_p_derived:
521            if (rel.distance == distance_min):
522                if rel.relation[1].category == ROOTED:
523                    return self.dataset.root
524                if rel.relation[1].category in (MIXED, ROOTDERIVED):
525                    return rel.relation[1]
526        return self.dataset.root

530        '''return the parent with minimal distance of the AnaDfield'''
531        if self.category in (UNIQUE, ROOTED, COUPLED):
532            return self.p_derived
533        dist_up = [rel.distance for rel in self.list_relations
534                   if rel.relation[1].lencodec >= self.lencodec 
535                   and rel.relation[1].category != COUPLED]
536        if not dist_up or min(dist_up) == self.dist_root:
537            return self.dataset.root
538        distance_min = min(dist_up) 
539        list_dmin = [rel.relation[1] for rel in self.list_relations
540                     if rel.distance == distance_min]
541        max_lencodec = max([fld.lencodec for fld in list_dmin])

544    def list_parents(self, typeparent='derived', mode='field'):
545        ''' return the list of the AnaDfield's parents in the family tree up to 
546        the root AnaDfield.
547

555        are represented with index, idfield, or object

557        parent = self
558        listparent = []
559        while parent != self.dataset.root:
560            parent = parent.p_derived if typeparent == 'derived' else parent.p_distance
561            if parent != self.dataset.root:

565    def dic_inner_node(self, mode, lname): 
566        '''return a child AnaDfield tree.
567
568         *Parameters*
569
570        - **lname** : integer - maximal length of the names       
571        - **mode** : str (default 'field') - AnaDfield representation 
572        ('field', 'id', 'index')     
573        
574        *Returns* : dict where key is a AnaDfield and value is the list of 
575        the childs.
576        '''
577        adding = ''
578        if mode == 'distance':
579            rel_parent = self.dataset.get_relation(self, self.p_distance)
580            adding = str(rel_parent.distance) + ' - '
581        elif mode == 'derived':
582            rel_parent = self.dataset.get_relation(self, self.p_derived)
583            adding = str(rel_parent.distance) + ' - '            
584        adding += str(self.lencodec)
585        name = self.idfield[:lname] + ' (' + adding + ')'
586        lis = [name.replace(' ', '*').replace("'", '*')]
587        if self.category != COUPLED:
588            for rel in self.list_coupled:
589                lis.append(rel.relation[1].dic_inner_node(mode, lname))            
590        if not self.category in (ROOTED, UNIQUE):
591            if mode == 'distance':
592                childs = [rel.relation[1] for rel in self.list_relations 
593                          if rel.relation[1].p_distance == self and 
594                             rel.relation[1].category != COUPLED]
595            elif mode == 'derived':
596                childs = [rel.relation[1] for rel in self.list_relations 
597                          if rel.relation[1].p_derived == self and 
598                             rel.relation[1].category != COUPLED]
599            for fld in childs:
600                lis.append(fld.dic_inner_node(mode, lname))            
601        return {str(self.index).ljust(2, '*'): lis}

784                   and rel.relation[1].index > rel.relation[0].index
785                   and rel.relation[0].category != COUPLED
786                   and rel.relation[1].category != COUPLED]
787        if distributed:
788            crossed = [rel for rel in crossed if rel.distrib]
789        if not crossed:
790            return []

792        if len(crossed) == 1 and crossed[0].dist == len(self):
793            partit.insert(0, crossed[0].relation)
794        else:
795            for repeat in list(range(len(crossed))):
796                candidates = combinations(crossed, repeat + 1)
797                for candidat in candidates:
798                    flds = list(set([rel.relation[i] for rel in candidat for i in [0,1]]))
799                    if (reduce(mul,[fld.lencodec for fld in flds]) == len(self) and
800                        len(candidat) == sum(range(len(flds))) and 
801                        (not distributed or min([rel.distrib for rel in candidat]))):
802                        partit.insert(0, flds)
803        partit = Util.view(partit, mode)
804        return [list(tup) for tup in sorted(sorted(list({tuple(sorted(prt))
805                          for prt in partit})), key=len, reverse=True)]
