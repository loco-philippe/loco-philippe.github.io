139    def encode_coef(lis):
140        '''Generate a repetition coefficient for periodic list'''
141        if len(lis) < 2:
142            return 0
143        coef = 0
144        for i in range(1,len(lis)):
145            coef = i
146            if lis[i-1] != lis[i]:
147                break
148        if lis == FieldInterface._periodic_keys(coef, max(lis) + 1, len(lis)):
149            return coef
150        return 0

154        ''' return a list of keys with periodic structure'''
