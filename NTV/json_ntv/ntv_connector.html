 35def from_csv(file_name, single_tab=True, dialect='excel', **fmtparams):
 36    ''' return a 'tab' NtvSingle from a csv file
 37
 38    *parameters*
 39
 40    - **file_name** : name of the csv file
 41    - **single_tab** : boolean (default True) - if True return a 'tab' NtvSingle,
 42    else return a NtvSet.
 43    - **dialect, fmtparams** : parameters of csv.DictReader object'''
 44    with open(file_name, newline='') as csvfile:
 45        reader = csv.DictReader(csvfile, dialect=dialect, **fmtparams)
 46        names = reader.fieldnames
 47        list_ntv_value = [[] for nam in names]
 48        for row in reader:
 49            for ind_field, val in enumerate(list(row.values())):
 50                list_ntv_value[ind_field].append(json.loads(val))
 51    list_ntv = []
 52    for ind_field, field in enumerate(names):
 53        list_ntv.append(
 54            NtvList(list_ntv_value[ind_field], *Ntv.from_obj_name(field)[:2]))
 55    if single_tab:
 56        return NtvSingle(NtvList(list_ntv, None, None).to_obj(), None, 'tab')
 57    return NtvList(list_ntv, None, None)

 60def to_csv(file_name, ntv, *args, restval='', extrasaction='raise', dialect='excel', **kwds):
 61    ''' convert a 'tab' NtvSingle into csv file and return the file name
 62
 63    *parameters*
 64
 65    - **file_name** : name of the csv file
 66    - **ntv** : 'tab' NtvSingle to convert
 67    - **args, restval, extrasaction, dialect, kwds** : parameters of csv.DictWriter object'''
 68    if isinstance(ntv, NtvSingle):
 69        ntv_set = Ntv.obj(ntv.ntv_value)
 70    else:
 71        ntv_set = ntv
 72    list_ntv = [Ntv.obj(field) for field in ntv_set]
 73    fieldnames = [ntv_field.json_name(string=True) for ntv_field in list_ntv]
 74    with open(file_name, 'w', newline='') as csvfile:
 75        writer = csv.DictWriter(csvfile, fieldnames=fieldnames, restval=restval,
 76                                extrasaction=extrasaction, dialect=dialect, *args, **kwds)
 77        writer.writeheader()
 78        for i in range(len(list_ntv[0])):
 79            writer.writerow({name: field_ntv[i].to_obj(field_ntv.ntv_type, encoded=True)
 80                             for name, field_ntv in zip(fieldnames, list_ntv)})
 81    return file_name

354    @staticmethod

356        ''' convert json ntv_value into a Series.
357
358        *Parameters*
359
360        - **index** : list (default None) - list of index values,
361        - **leng** : integer (default None) - leng of the series
362        - **alias** : boolean (default False) - if True dtype alias else default dtype
363        - **annotated** : boolean (default False) - if True, NTV names are not included.'''
364        option = {'index': None, 'leng': None, 'alias': False,
365                  'annotated': False} | kwargs
366        types = SeriesConnec.types
367        astype = SeriesConnec.astype
368        deftype = SeriesConnec.deftype
369        #ntv = Ntv.obj(ntv_value)
370        ntv = Ntv.fast(ntv_value)
371        ntv_name = ntv.name
372        codes = len(ntv) == 2 and len(ntv[1]) > 1  # categorical
373        if codes:
374            cod = ntv[1].to_obj(simpleval=True, fast=True)
375            #cod = ntv[1].to_obj(simpleval=True)
376            ntv = ntv[0]
377        ntv_type = ntv.type_str if ntv.type_str != 'json' else ''
378        pd_convert = ntv_type in types['ntv_type'].values or ntv_type == ''
379        dtype = 'object'
380        if pd_convert:  # pandas conversion
381            dtype = types.set_index(
382                'ntv_type').loc[ntv_type]['dtype'] if ntv_type != '' else None
383
384        # calcul name_type, pd_name, ntv_obj
385        if pd_convert:
386            name_type = types.set_index(
387                'ntv_type').loc[ntv_type]['name_type'] if ntv_type != '' else ''
388            pd_name = ntv_name + '::' + name_type if name_type else ntv_name
389            pd_name = pd_name if pd_name else None
390            if name_type == 'array':
391                ntv_obj = ntv.to_obj(format='obj', simpleval=True)
392            else:
393                ntv_obj = ntv.obj_value(simpleval=option['annotated'],
394                                        json_array=False, def_type=ntv.type_str)
395                ntv_obj = ntv_obj if isinstance(ntv_obj, list) else [ntv_obj]
396        else:
397            name_type = ntv_type
398            pd_name = ntv_name+'::'+name_type
399            ntv_obj = ntv.to_obj(
400                format='obj', simpleval=True, def_type=ntv_type)
401            # ntv_obj = ntv.to_obj_ntv(simpleval=True, def_type=ntv_type)  #!!!
402
403        # calcul de srs
404        if codes:
405            if pd_convert and name_type != 'array':
406                categories = pd.read_json(json.dumps(
407                    ntv_obj), dtype=dtype, typ='series')
408                cat_type = categories.dtype.name
409                categories = categories.astype(astype.get(cat_type, cat_type))
410            else:
411                categories = pd.Series(ntv_obj, dtype='object')
412            cat = pd.CategoricalDtype(categories=categories)
413            data = pd.Categorical.from_codes(codes=cod, dtype=cat)
414            srs = pd.Series(data, name=pd_name,
415                           index=option['index'], dtype='category')
416        else:
417            data = ntv_obj * \
418                option['leng'] if len(ntv) == 1 and option['leng'] else ntv_obj
419            if pd_convert:
420                srs = pd.read_json(json.dumps(data), dtype=dtype,
421                                  typ='series').rename(pd_name)
422            else:
423                srs = pd.Series(data, name=pd_name, dtype=dtype)
424
425        if option['alias']:
426            return srs.astype(astype.get(srs.dtype.name, srs.dtype.name))
427        return srs.astype(deftype.get(srs.dtype.name, srs.dtype.name))

452            else:

454                    orient='records', date_format='iso', default_handler=str))
