206    def __init__(self, listidx=None, name=None, reindex=True):
207        '''
208        Dataset constructor.
209
210        *Parameters*
211
212        - **listidx** :  list (default None) - list of Field data
213        - **name** :  string (default None) - name of the dataset
214        - **reindex** : boolean (default True) - if True, default codec for each Field'''
215
216        if isinstance(listidx, Cdataset):
217            self.lindex = [copy(idx) for idx in listidx.lindex]
218            self.name = name if name else listidx.name
219            self._analysis = listidx._analysis
220            return
221        if listidx.__class__.__name__ == 'DataFrame':
222            lindex = NtvConnector.connector(
223            )['DataFrameConnec'].to_listidx(listidx)[0]
224            listidx = [Cfield(field['codec'], field['name'], field['keys'])
225                       for field in lindex]
226        self.name = name
227        self.lindex = [] if listidx is None else listidx
228        if reindex:
229            self.reindex()
230        self._analysis = None
231        return

346    @classmethod
347    def from_ntv(cls, ntv_value, reindex=True, decode_str=False, fast=False):
348        '''Generate a Dataset Object from a ntv_value
349
350        *Parameters*
351
352        - **ntv_value** : bytes, string, Ntv object to convert
353        - **reindex** : boolean (default True) - if True, default codec for each Field
354        - **decode_str**: boolean (default False) - if True, string are loaded in json data
355        - **fast** : boolean (default False) - if True, ntv_value are not converted in json-value'''
356        ntv = Ntv.obj(ntv_value, decode_str=decode_str, fast=fast)
357        if len(ntv) == 0:
358            return cls()
359        lidx = [list(NtvUtil.decode_ntv_tab(
360            ntvf, cls.field_class.ntv_to_val)) for ntvf in ntv]
361        leng = max(idx[6] for idx in lidx)
362        for ind in range(len(lidx)):
363            if lidx[ind][0] == '':
364                lidx[ind][0] = 'i'+str(ind)
365            NtvConnector.init_ntv_keys(ind, lidx, leng)
366        lindex = [cls.field_class(idx[2], idx[0], idx[4], None,  # idx[1] pour le type,
367                                  reindex=reindex) for idx in lidx]
368        return cls(lindex, reindex=reindex, name=ntv.name)

370    def add(self, other, name=False, solve=True):
371        ''' Add other's values to self's values for each index
372
373        *Parameters*
374
375        - **other** : Dataset object to add to self object
376        - **name** : Boolean (default False) - Add values with same index name (True) or
377        same index row (False)
378        - **solve** : Boolean (default True) - If True, replace None other's codec value
379        with self codec value.
380
381        *Returns* : self '''
382        if self.lenindex != other.lenindex:
383            raise DatasetError('length are not identical')
384        if name and sorted(self.lname) != sorted(other.lname):
385            raise DatasetError('name are not identical')
386        for i in range(self.lenindex):
387            if name:
388                self.lindex[i].add(other.lindex[other.lname.index(self.lname[i])],
389                                   solve=solve)
390            else:
391                self.lindex[i].add(other.lindex[i], solve=solve)
392        return self

394    def to_analysis(self, distr=False):
395        '''return a dict with data used in AnaDataset module
396
397        *Parameters*
398
399        - **distr** : Boolean (default False) - If True, add distr information'''
400        return {'name': self.name, 'fields': [fld.to_analysis for fld in self.lindex],
401                'length': len(self), 'hashd': self._hashd,
402                'relations': {self.lindex[i].name:
403                              {self.lindex[j].name: Cutil.dist(
404                                  self.lindex[i].keys, self.lindex[j].keys, distr)
405                               for j in range(i+1, len(self.lindex))}
406                              for i in range(len(self.lindex)-1)}
407                }

415    def delindex(self, delname=None, savename=None):
416        '''remove an Field or a list of Field.
417
418        *Parameters*
419
420        - **delname** : string or list of string - name of index to remove
421        - **savename** : string or list of string - name of index to keep
422
423        *Returns* : none '''
424        if not delname and not savename:
425            return
426        if isinstance(delname, str):
427            delname = [delname]
428        if isinstance(savename, str):
429            savename = [savename]
430        if delname and savename:
431            delname = [name for name in delname if not name in savename]
432        if not delname:
433            delname = [name for name in self.lname if not name in savename]
434        for idxname in delname:
435            if idxname in self.lname:
436                self.lindex.pop(self.lname.index(idxname))

438    def nindex(self, name):
439        ''' index with name equal to attribute name'''
440        if name in self.lname:
441            return self.lindex[self.lname.index(name)]
442        return None

444    def renameindex(self, oldname, newname):
445        '''replace an index name 'oldname' by a new one 'newname'. '''
446        for i in range(self.lenindex):
447            if self.lname[i] == oldname:
448                self.lindex[i].setname(newname)
449        for i in range(len(self.lvarname)):
450            if self.lvarname[i] == oldname:
451                self.lvarname[i] = newname
452
453    def reorder(self, recorder=None):
454        '''Reorder records in the order define by 'recorder' '''
455        if recorder is None or set(recorder) != set(range(len(self))):
456            return None
457        for idx in self.lindex:
458            idx.set_keys([idx.keys[i] for i in recorder])
459        return None
460
461    def setname(self, listname=None):
462        '''Update Field name by the name in listname'''
463        for i in range(min(self.lenindex, len(listname))):
464            self.lindex[i].name = listname[i]

466    def swapindex(self, order):
467        '''
468        Change the order of the index .
469
470        *Parameters*
471
472        - **order** : list of int or list of name - new order of index to apply.
473
474        *Returns* : self '''
475        if self.lenindex != len(order):
476            raise DatasetError('length of order and Dataset different')
477        if not order or isinstance(order[0], int):
478            self.lindex = [self.lindex[ind] for ind in order]
479        elif isinstance(order[0], str):
480            self.lindex = [self.nindex(name) for name in order]
481        return self

483    def check_relation(self, field, parent, typecoupl, value=True):
484        '''get the inconsistent records for a relationship
485
486         *Parameters*
487
488        - **field** : int or str - index or name of the field involved in the relation
489        - **parent**: int or str - index or name of the second field involved in the relation
490        - **typecoupl**: str - relationship to check ('derived' or 'coupled')
491        - **value**: boolean (default True) - if True return a dict with inconsistent
492        values of the fields, else a tuple with index of records)
493
494        *Returns* :
495
496        - dict with inconsistent values of the fields
497        - or a tuple with index of records'''
498        f_parent = self.nindex(parent) if isinstance(
499            parent, str) else self.lindex[parent]
500        f_field = self.nindex(field) if isinstance(
501            field, str) else self.lindex[field]
502        match typecoupl:
503            case 'derived':
504                errors = f_parent.coupling(f_field, reindex=True)
505            case 'coupled':
506                errors = f_parent.coupling(
507                    f_field, derived=False, reindex=True)
508            case _:
509                raise DatasetError(typecoupl + "is not a valid relationship")
510        if not value:
511            return errors
512        return {'row': list(errors), f_field.name: f_field[errors], f_parent.name: f_parent[errors]}
513
514    def check_relationship(self, relations):
515        '''get the inconsistent records for each relationship defined in relations
516
517         *Parameters*
518
519        - **relations** : list of dict or single dict - list of fields with relationship property
520
521        *Returns* :
522
523        - dict with for each relationship: key = string with the two fields name,
524        and value = list of inconsistent records
525        - or if single relationship : value'''
526        if not isinstance(relations, (list, dict)):
527            raise DatasetError("relations is not correct")
528        if isinstance(relations, dict):
529            relations = [relations]
530        dic_res = {}
531        for field in relations:
532            if not 'relationship' in field or not 'name' in field:
533                continue
534            if not 'parent' in field['relationship'] or not 'link' in field['relationship']:
535                raise DatasetError("relationship is not correct")
536            rel = field['relationship']['link']
537            f_parent = field['relationship']['parent']
538            f_field = field['name']
539            name_rel = f_field + ' - ' + f_parent
540            if self.nindex(f_parent) is None or self.nindex(f_field) is None:
541                raise DatasetError("field's name is not present in data")
542            dic_res[name_rel] = self.check_relation(
543                f_field, f_parent, rel, False)
544        if len(dic_res) == 1:
545            return list(dic_res.values())[0]
546        return dic_res
