 85    def coupling(self, idx, derived=True):
 86        '''
 87        Transform indexes in coupled or derived indexes (codec extension).
 88        If derived option is True, self._codec is extended and idx codec not,
 89        else, both are coupled and both codec are extended.
 90
 91        *Parameters*
 92
 93        - **idx** : single Iindex or list of Iindex to be coupled or derived.
 94        - **derived** : boolean (default : True)
 95
 96        *Returns* : tuple with duplicate records (errors)'''
 97        if not isinstance(idx, list):
 98            index = [idx]
 99        else:
100            index = idx
101        idxzip = self.__class__.ext(list(zip(*([self._keys] + [ix._keys for ix in index]))),
102                                     typevalue=None)
103        self.tocoupled(idxzip)
104        if not derived:
105            for ind in index:
106                ind.tocoupled(idxzip)

109    def couplinginfos(self, other, default=False):
110        '''return a dict with the coupling info between other (lencoupling, rate,
111        disttomin, disttomax, distmin, distmax, diff, typecoupl)
112
113        *Parameters*
114
115        - **other** : other index to compare
116        - **default** : comparison with default codec
117
118        *Returns* : dict'''
119        if default:
120            return util.couplinginfos(self.values, other.values)
121        if min(len(self), len(other)) == 0:
122            return {'lencoupling': 0, 'rate': 0, 'disttomin': 0, 'disttomax': 0,
123                    'distmin': 0, 'distmax': 0, 'diff': 0, 'typecoupl': 'null'}
124        lens = len(self._codec)
125        leno = len(other._codec)
126        xmin = max(lens, leno)
127        xmax = lens * leno
128        diff = abs(lens - leno)
129        if min(lens, leno) == 1:
130            if lens == 1:
131                typec = 'derived'
132            else:
133                typec = 'derive'
134            return {'lencoupling': xmin, 'rate': 0, 'disttomin': 0, 'disttomax': 0,
135                    'distmin': xmin, 'distmax': xmax, 'diff': diff, 'typecoupl': typec}
136        xso = len(util.tocodec([tuple((v1, v2))
137                  for v1, v2 in zip(self._keys, other._keys)]))
138        dic = {'lencoupling': xso, 'rate': (xso - xmin) / (xmax - xmin),
139               'disttomin': xso - xmin,  'disttomax': xmax - xso,
140               'distmin': xmin, 'distmax': xmax, 'diff': diff}
141        if dic['rate'] == 0 and dic['diff'] == 0:
142            dic['typecoupl'] = 'coupled'
143        elif dic['rate'] == 0 and lens < leno:
144            dic['typecoupl'] = 'derived'
145        elif dic['rate'] == 0 and lens > leno:
146            dic['typecoupl'] = 'derive'
147        elif dic['rate'] == 1:
148            dic['typecoupl'] = 'crossed'
149        elif lens < leno:
150            dic['typecoupl'] = 'linked'
151        else:
152            dic['typecoupl'] = 'link'
153        return dic

201    def getduplicates(self):
202        ''' return tuple of items with duplicate codec'''
203        count = Counter(self._codec)
204        defcodec = list(count - Counter(list(count)))
205        dkeys = defaultdict(list)
206        for key, ind in zip(self._keys, range(len(self))):
207            dkeys[key].append(ind)
208        dcodec = defaultdict(list)
209        for key, ind in zip(self._codec, range(len(self._codec))):
210            dcodec[key].append(ind)
211        duplicates = []
212        for item in defcodec:
213            for codecitem in dcodec[item]:
214                duplicates += dkeys[codecitem]
215        return tuple(duplicates)
