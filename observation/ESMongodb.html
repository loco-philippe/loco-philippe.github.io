126    def _cond(self, operand, comp, path, all = True):
127        """
128        Prend en entrée des paramètres et retourne un dictionnaire de la condition pour la recherche MongoDB.
129        Tri par dates : (Tout) / (au moins une date) est (supérieur à) / (inférieur à) / (égal à) / (dans) [date(s) en paramètre]
130        (syntaxe à vérifier) -> Dans le cas d'un intervalle, cette fonction est appelée plusieurs fois.
131        -> Traiter le cas de conditions redondantes (ex <4 => <8) (actuelleme,t, le dernier arrivé l'emporte)
132        -> Max actuel : 4 conditions : "tout" dans un intervalle et "au moins un" dans un intervalle inclus dans celui-ci.
133        L'utilisation de OR est gérée par _search.
134        Format dates : array de 2^k valeurs, kcN [date1, date2, date3, date3] <-> Timeslot([[date1, date2], date3])
135        """
136        
137        if isinstance(comp, str) and comp[0] != "$":
138            if comp in {"eq", "=", "=="}        : comp = "$eq"
139            elif comp in {"gte", ">=", "=>"}    : comp = "$gte"
140            elif comp in {"gt", ">"}            : comp = "$gt"
141            elif comp in {"lte", "<=", "=<"}    : comp = "$lte"
142            elif comp in {"lt", "<"}            : comp = "$lt"
143            elif comp == "in"                   : comp = "$in"
144            else:
145                raise ValueError("Comparators allowed are =, <, >, <=, >=, in and MongoDB equivalents.")
146        
147        if all:
148            if comp == "$eq"    :   cond_0 = {"$nor" : [{"$lt" : operand}, {"$gt" : operand}]}
149            elif comp == "$gte" :   cond_0 = {"$not" : {"$lt"  : operand}}
150            elif comp == "$gt"  :   cond_0 = {"$not" : {"$lte" : operand}}
151            elif comp == "$lte" :   cond_0 = {"$not" : {"$gt"  : operand}}
152            elif comp == "$lt"  :   cond_0 = {"$not" : {"$gte" : operand}}
153        else:                                 cond_0 = {comp : operand}
154
155        if self.searchtype == "aggregate":
156            if path not in self._match_2:
157                self._match_2[path] = cond_0
158            else:
159                self._match_2[path] |= cond_0
160        
161        return { path : cond_0 }

191            self._request.append({"$project" : self._project})

193            self._request.append({"$sort" : self._sort})
