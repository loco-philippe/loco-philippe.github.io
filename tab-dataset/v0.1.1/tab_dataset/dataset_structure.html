 69    def addindex(self, index, first=False, merge=False, update=False):
 70        '''add a new index.
 71
 72        *Parameters*
 73
 74        - **index** : Field - index to add (can be index Ntv representation)
 75        - **first** : If True insert index at the first row, else at the end
 76        - **merge** : create a new index if merge is False
 77        - **update** : if True, update actual values if index name is present (and merge is True)

 86                self.lindex.insert(0, idx)

 91                idx.name += '(2)'

 94            else:
 95                self.lindex.append(idx)
 96        elif update:  # si merge et si idx.name in idxname
