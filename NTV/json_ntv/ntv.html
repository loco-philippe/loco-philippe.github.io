NTV.json_ntv.ntv

Created on Feb 27 2023

@author: Philippe@loco-labs.io

The ntv module is part of the NTV.json_ntv package (specification document).

It contains the classes NtvSingle, NtvList, Ntv(abstract), NtvConnector, NtvTree and NtvError for NTV entities.

1 - JSON-NTV structure

The NTV triplet (name, type, value) is represented using a JSON-NTV format inspired by the RFC JSON-ND project :

  • value (if name and type are not documented)
  • { "name" : value } (if name is documented but not type)
  • { ":type" : value } for primitive entities and { "::type" : value } for structured entities (if type is documented but not name)
  • { "name:type" : value } for primitive entities and { "name::type" : value } for structured entities (if type and name are documented).

For an NTV-single, the value is the JSON-value of the entity. For an NTV-list, value is a JSON-array (JSON-object) where JSON-elements (JSON-members) are the JSON-NTV formats of included NTV entities.

This JSON-NTV format allows full compatibility with existing JSON structures:

  • a JSON-number, JSON-string or JSON-boolean is the representation of an NTV-single entity,
  • a JSON-object with a single member is the representation of an NTV-single entity
  • a JSON-array or JSON-object is the representation of an NTV-list entity

174    @staticmethod
175    def from_obj(value, def_type=None, def_sep=None, no_typ=False):
176        ''' return an Ntv entity from an object value.
177
178        *Parameters*
179
180        - **value**: Ntv value to convert in an Ntv entity
181        - **no_typ** : boolean (default None) - if True, NtvList is with 'json' type
182        - **def_type** : NtvType or Namespace (default None) - default type of the NTV entity
183        - **def_sep**: ':', '::' or None (default None) - default separator of the Ntv entity'''
184        value = Ntv._from_value(value)
185        #notype = def_type == 'notype'
186        if value.__class__.__name__ in ['NtvSingle', 'NtvList']:
187            return value
188        ntv_name, str_typ, ntv_value, sep = Ntv._decode(value)
189        sep = def_sep if not sep else sep
190        if isinstance(ntv_value, list) and sep in (None, '::'):
191            def_type = agreg_type(str_typ, def_type, False)
192            sep = None if sep and not def_type else sep
193            sep = ':' if sep else sep
194            ntv_list = [Ntv.from_obj(val, def_type, sep) for val in ntv_value]
195            if not def_type and ntv_list:
196                def_type = ntv_list[0].ntv_type
197            def_type = 'json' if no_typ else def_type
198            return NtvList(ntv_list, ntv_name, def_type)
199        if sep == ':' or (sep is None and isinstance(ntv_value, dict) and
200                          len(ntv_value) == 1):
201            ntv_type = agreg_type(str_typ, def_type, False)
202            return NtvSingle(ntv_value, ntv_name, ntv_type)
203        if sep is None and not isinstance(ntv_value, dict):
204            is_json = isinstance(value, (int, str, float, bool))
205            ntv_type = agreg_type(str_typ, def_type, is_json)
206            return NtvSingle(ntv_value, ntv_name, ntv_type)
207        if isinstance(ntv_value, dict) and (sep == '::' or len(ntv_value) != 1 and
208                                            sep is None):
209            keys = list(ntv_value.keys())
210            values = list(ntv_value.values())
211            def_type = agreg_type(str_typ, def_type, False)
212            sep = None if sep and not def_type else sep
213            sep = ':' if sep else sep
214            ntv_list = [Ntv.from_obj({key: val}, def_type, sep)
215                        for key, val in zip(keys, values)]
216            if not def_type and ntv_list:
217                def_type = ntv_list[0].ntv_type
218            def_type = 'json' if no_typ else def_type
219            return NtvList(ntv_list, ntv_name, def_type)
220        raise NtvError('separator ":" is not compatible with value')

319        '''return a tuple with name, type and separator from string'''
320        if not isinstance(string, str):
321            raise NtvError('a json-name have to be str')
322        if string == '':
323            return (None, None, None)

325        if '::' in string:
326            sep = '::'
327        elif ':' in string:
328            sep = ':'
329        if sep is None:
330            return (string, None, None)
331        split = string.rsplit(sep, 2)
332        if len(split) == 1:
333            return (string, None, sep)
334        if split[0] == '':

336        if split[1] == '':
337            return (split[0], None, sep)

425    def to_repr(self, nam=True, typ=True, val=True, maxi=10):
426        '''return a simple json representation of the Ntv entity.
427
428        *Parameters*
429
430        - **nam**: Boolean (default True) : if true, the names are included
431        - **typ**: Boolean (default True) : if true, the types are included
432        - **val**: Boolean (default True) : if true, the values are included
433        - **maxi**: Integer (default 10) : number of values to included for NtvList
434        entities. If maxi < 1 all the values are included.
435        '''
436        ntv = self.code_ntv
437        if nam and typ:
438            ntv = ntv[0]
439        if self.ntv_name and nam:
440            ntv += '-' + self.ntv_name
441        if self.ntv_type and typ:
442            ntv += '-' + self.ntv_type.long_name
443        if isinstance(self, NtvSingle) and not isinstance(self.ntv_value, NtvSingle):
444            if val:
445                if ntv:
446                    ntv += '-'
447                ntv += json.dumps(self.ntv_value)
448            return ntv
449        if isinstance(self, NtvSingle) and isinstance(self.ntv_value, NtvSingle):
450            return {ntv:  self.ntv_value.to_repr(nam, typ, val)}
451        if isinstance(self, NtvList):
452            if maxi < 1:
453                maxi = len(self.ntv_value)
454            return {ntv:  [ntvi.to_repr(nam, typ, val) for ntvi in self.ntv_value[:maxi]]}
455        raise NtvError('the ntv entity is not consistent')

613    def _uncast(self, **option):
614        '''return object from ntv_value'''
615        dic_fct = {'date': datetime.date.fromisoformat, 'time': datetime.time.fromisoformat,
616                   'datetime': datetime.datetime.fromisoformat, 'array': tuple}
617        dic_geo = {'point': 'point', 'multipoint': 'multipoint', 'line': 'linestring',
618                   'multiline': 'multilinestring', 'polygon': 'polygon',
619                   'multipolygon': 'multipolygon'}
620        dic_cbor = {'point': False, 'multipoint': False, 'line': False,
621                    'multiline': False, 'polygon': False, 'multipolygon': False,
622                    'date': True, 'time': False, 'datetime': True}
623        dic_obj = {'tab': 'IlistConnec',
624                   'field': 'IindexConnec', 'other': None}
625        type_n = self.ntv_type.name
626        if 'dicobj' in option:
627            dic_obj |= option['dicobj']
628        obj = not option['encode_format'] == 'cbor' or \
629            (self.ntv_type and type_n in dic_cbor and dic_cbor[type_n])
630        if self.ntv_type is None or not obj:
631            return self.ntv_value
632        if type_n in dic_fct:
633            return dic_fct[type_n](self.ntv_value)
634        if type_n == 'ntv':
635            return Ntv.obj(self.ntv_value)
636        if type_n in dic_geo:
637            return geometry.shape({"type": dic_geo[type_n],
638                                   "coordinates": self.ntv_value})
639        connec = None
640        if type_n in dic_obj and \
641                dic_obj[type_n] in NtvConnector.connector():
642            connec = NtvConnector.connector()[dic_obj[type_n]]
643        elif dic_obj['other'] in NtvConnector.connector():
644            connec = NtvConnector.connector()['other']
645        if connec:
646            return connec.from_ntv(self.ntv_value, **option)
647        return self.ntv_value
