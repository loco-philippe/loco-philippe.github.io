259                 optcsv={'quoting': csv.QUOTE_NONNUMERIC}, dtype=ES.def_dtype):
260        '''
261        Ilist constructor (from a csv file). Each column represents index values.
262
263        *Parameters*
264
265        - **filename** : string (default 'ilist.csv'), name of the file to read
266        - **header** : boolean (default True). If True, the first raw is dedicated to names
267        - **nrow** : integer (default None). Number of row. If None, all the row else nrow
268        - **dtype** : list of string (default None) - data type for each column (default str)
269        - **optcsv** : dict (default : quoting) - see csv.reader options'''
270        if not optcsv:
271            optcsv = {}
272        if not nrow:
273            nrow = -1
274        with open(filename, newline='', encoding="utf-8") as file:
275            reader = csv.reader(file, **optcsv)
276            irow = 0
277            for row in reader:
278                if irow == nrow:
279                    break
280                if irow == 0:
281                    if dtype and not isinstance(dtype, list):
282                        dtype = [dtype] * len(row)
283                    idxval = [[] for i in range(len(row))]
284                    idxname = None
285                if irow == 0 and header:
286                    idxname = row
287                else:
288                    if not dtype:
289                        for i in range(len(row)):
290                            idxval[i].append(row[i])
291                    else:
292                        for i in range(len(row)):
293                            idxval[i].append(util.cast(row[i], dtype[i]))
294                irow += 1
295        return cls.ext(idxval, idxname, typevalue=None, reindex=True)

355    def merge(self, fillvalue=math.nan, reindex=False, simplename=False):
356        '''
357        Merge method replaces Ilist objects included into its constituents.
358
359        *Parameters*
360
361        - **fillvalue** : object (default nan) - value used for the additional data
362        - **reindex** : boolean (default False) - if True, set default codec after transformation
363        - **simplename** : boolean (default False) - if True, new Iindex name are
364        the same as merged Iindex name else it is a composed name.
365
366        *Returns*: merged Ilist '''
367        ilc = copy(self)
368        delname = []
369        row = ilc[0]
370        if not isinstance(row, list):
371            row = [row]
372        merged, oldname, newname = Ilist._mergerecord(Ilist.ext(row, ilc.lname),
373                                                      simplename=simplename)
374        if oldname and not oldname in merged.lname:
375            delname.append(oldname)
376        for ind in range(1, len(ilc)):
377            oldidx = ilc.nindex(oldname)
378            for name in newname:
379                ilc.addindex(Iindex(oldidx.codec, name, oldidx.keys))
380            row = ilc[ind]
381            if not isinstance(row, list):
382                row = [row]
383            rec, oldname, newname = Ilist._mergerecord(Ilist.ext(row, ilc.lname),
384                                                       simplename=simplename)
385            if oldname and newname != [oldname]:
386                delname.append(oldname)
387            for name in newname:
388                oldidx = merged.nindex(oldname)
389                fillval = util.castval(
390                    fillvalue, util.typename(name, ES.def_clsName))
391                merged.addindex(
392                    Iindex([fillval] * len(merged), name, oldidx.keys))
393            merged += rec
394        for name in set(delname):
395            if name:
396                merged.delindex(name)
397        if reindex:
398            merged.reindex()
399        ilc.lindex = merged.lindex
400        return ilc

403    @classmethod
404    def _init_obj(cls, listidx=None, reindex=True, typevalue=ES.def_clsName, context=True):
405        '''
406        Ilist constructor.
407
408        *Parameters*
409
410        - **listidx** :  list (default None) - list of compatible Iindex data
411        - **reindex** : boolean (default True) - if True, default codec for each Iindex
412        - **typevalue** : str (default ES.def_clsName) - default value class (None or NamedValue)
413        '''
414        lindex = []
415        if listidx.__class__.__name__ == 'DataFrame':
416            lindex = []
417            for name, idx in listidx.astype('category').items():
418                lis = list(idx.cat.categories)
419                if lis and isinstance(lis[0], pandas._libs.tslibs.timestamps.Timestamp):
420                    lis = [ts.to_pydatetime().astimezone(datetime.timezone.utc)
421                           for ts in lis]
422                lindex.append(Iindex(lis, name, list(idx.cat.codes),
423                                     lendefault=len(listidx), castobj=False))
424            return cls(lindex, reindex=reindex)
425
426        if isinstance(listidx, dict):
427            for idxname in listidx:
428                var, idx = Iindex.from_dict_obj({idxname: listidx[idxname]},
429                                                typevalue=typevalue, reindex=reindex)
430                lindex.append(idx)
431            return cls(lindex, reindex=reindex)
432
433        if isinstance(listidx, list) and len(listidx) == 0:
434            return cls()
435
436        # decode: name, typevaluedec, codec, parent, keys, isfullkeys, isparent, isvar
437        lidx = [list(IindexInterface.decodeobj(idx, typevalue, context))
438                for idx in listidx]
439        for ind in range(len(lidx)):
440            if lidx[ind][0] is None or lidx[ind][0] == ES.defaultindex:
441                lidx[ind][0] = 'i'+str(ind)
442            if lidx[ind][1] is None:
443                lidx[ind][1] = util.typename(lidx[ind][0], typevalue)
444        name, typevaluedec, codec, parent, keys, isfullkeys, isparent, isvar =\
445            tuple(zip(*lidx))
446
447        leng = [len(cod) for cod in codec]
448        # mode full : tous False et même longueur
449        fullmode = not max(isfullkeys) and max(leng) == min(leng)
450        # mode default : tous True (idem all(isfullkeys))
451        defmode = min(isfullkeys)
452
453        # not max(isparent) : tous isparent = False
454        if not max(isparent) and (fullmode or defmode):  # mode full ou mode default
455            lindex = [Iindex(idx[2], idx[0], idx[4], idx[1],
456                             reindex=reindex) for idx in lidx]
457            return cls(lindex, reindex=reindex)
458
459        length, crossed = Ilist._init_len_cros(
460            fullmode, leng, isfullkeys, keys, isparent)
461        keyscross = util.canonorder([leng[i] for i in crossed])
462        # name: 0, typevaluedec: 1, codec: 2, parent: 3, keys: 4
463        for ind in range(len(crossed)):
464            lidx[crossed[ind]][4] = keyscross[ind]  # keys
465        for ind in range(len(lidx)):
466            Ilist._init_keys(ind, lidx, length)
467        lindex = [Iindex(idx[2], idx[0], idx[4], idx[1],
468                         reindex=reindex) for idx in lidx]
469        return cls(lindex, reindex=False)

471    @staticmethod
472    def _init_len_cros(fullmode, leng, isfullkeys, keys, isparent):
473        ''' initialization of length and crossed data'''
474        # crossed : pas d'index (isfullindex false), pas de parent(isparent false)
475        crossed = []
476        if fullmode:  # au moins un fullkeys ou une longueur différente
477            length = max(leng)
478            return length, crossed
479
480        # au moins un fullkeys ou une longueur différente
481        if max(isfullkeys):
482            length = len(keys[isfullkeys.index(True)])
483            crossed = [i for i, (isfullk, ispar, lengt) in
484                       enumerate(zip(isfullkeys, isparent, leng))
485                       if not ispar and not isfullk and 1 < lengt < length]
486            return length, crossed
487
488        # max(leng) != min(leng) pas de fullindex => matrice et dérivés
489        crossed = [i for i, (isfullk, ispar, lengt) in
490                   enumerate(zip(isfullkeys, isparent, leng))
491                   if not ispar and not isfullk and 1 < lengt]
492        lencrossed = [leng[ind] for ind in crossed]
493
494        if max(lencrossed) == min(lencrossed):
495            length = lencrossed[0]
496            crossed = []
497            return length, crossed
498
499        length = math.prod([leng[i] for i in crossed])
500        if length / max(lencrossed) == max(lencrossed):
501            length = max(lencrossed)
502            crossed = [i for i, (isfullk, ispar, lengt) in
503                       enumerate(zip(isfullkeys, isparent, leng))
504                       if not ispar and not isfullk and 1 < lengt < length]
505        return length, crossed

507    @staticmethod
508    def _init_keys(ind, lidx, leng):
509        ''' initialization of keys data'''
510        # name: 0, typevaluedec: 1, codec: 2, parent: 3, keys: 4
511        if lidx[ind][4] and (lidx[ind][3] is None or lidx[ind][3] < 0):
512            return
513        if lidx[ind][4] and len(lidx[ind][4]) == leng:
514            return
515        if len(lidx[ind][2]) == 1:
516            lidx[ind][4] = [0] * leng
517            return
518        if lidx[ind][3] is None:
519            raise IlistError('keys not referenced')
520        if lidx[ind][3] < 0:
521            lidx[ind][4] = list(range(leng))
522            return
523        if not lidx[lidx[ind][3]][4] or len(lidx[lidx[ind][3]][4]) != leng:
524            Ilist._init_keys(lidx[ind][3], lidx, leng)
525        if not lidx[ind][4] and len(lidx[ind][2]) == len(lidx[lidx[ind][3]][2]):
526            # coupled format
527            lidx[ind][4] = lidx[lidx[ind][3]][4]
528            return
529        # derived keys
530        if not lidx[ind][4]:  # derived format without keys
531            lenp = len(lidx[lidx[ind][3]][2])  # len codec parent
532            lidx[ind][4] = [(i*len(lidx[ind][2])) // lenp for i in range(lenp)]
533        lidx[ind][4] = Iindex.keysfromderkeys(
534            lidx[lidx[ind][3]][4], lidx[ind][4])
535        return

537    @staticmethod
538    def _mergerecord(rec, mergeidx=True, updateidx=True, simplename=False):
539        row = rec[0]
540        if not isinstance(row, list):
541            row = [row]
542        var = -1
543        for ind, val in enumerate(row):
544            if val.__class__.__name__ in ['Ilist', 'Observation']:
545                var = ind
546                break
547        if var < 0:
548            return (rec, None, [])
549        ilis = row[var]
550        oldname = rec.lname[var]
551        if ilis.lname == ['i0']:
552            newname = [oldname]
553            ilis.setname(newname)
554        elif not simplename:
555            newname = [oldname + '_' + name for name in ilis.lname]
556            ilis.setname(newname)
557        else:
558            newname = copy(ilis.lname)
559        for name in rec.lname:
560            if name in newname:
561                newname.remove(name)
562            else:
563                updidx = name in ilis.lname and not updateidx
564                ilis.addindex([name, [rec.nindex(name)[0]] * len(ilis)],
565                              merge=mergeidx, update=updidx)
566        return (ilis, oldname, newname)

393                    Iindex([fillval] * len(merged), name, oldidx.keys))

405    def _init_obj(cls, listidx=None, reindex=True, typevalue=ES.def_clsName, context=True):
406        '''
407        Ilist constructor.
408
409        *Parameters*
410
411        - **listidx** :  list (default None) - list of compatible Iindex data
412        - **reindex** : boolean (default True) - if True, default codec for each Iindex
413        - **typevalue** : str (default ES.def_clsName) - default value class (None or NamedValue)
414        '''
415        lindex = []
416        if listidx.__class__.__name__ == 'DataFrame':
417            lindex = []
418            for name, idx in listidx.astype('category').items():
419                lis = list(idx.cat.categories)
420                if lis and isinstance(lis[0], pandas._libs.tslibs.timestamps.Timestamp):
421                    lis = [ts.to_pydatetime().astimezone(datetime.timezone.utc)
422                           for ts in lis]
423                lindex.append(Iindex(lis, name, list(idx.cat.codes),
424                                     lendefault=len(listidx), castobj=False))
425            return cls(lindex, reindex=reindex)
426
427        if isinstance(listidx, dict):
428            for idxname in listidx:
429                var, idx = Iindex.from_dict_obj({idxname: listidx[idxname]},
430                                                typevalue=typevalue, reindex=reindex)
431                lindex.append(idx)
432            return cls(lindex, reindex=reindex)
433
434        if isinstance(listidx, list) and len(listidx) == 0:
435            return cls()
436
437        # decode: name, typevaluedec, codec, parent, keys, isfullkeys, isparent, isvar
438        lidx = [list(IindexInterface.decodeobj(idx, typevalue, context))
439                for idx in listidx]
440        for ind in range(len(lidx)):
441            if lidx[ind][0] is None or lidx[ind][0] == ES.defaultindex:
442                lidx[ind][0] = 'i'+str(ind)
443            if lidx[ind][1] is None:
444                lidx[ind][1] = util.typename(lidx[ind][0], typevalue)
445        name, typevaluedec, codec, parent, keys, isfullkeys, isparent, isvar =\
446            tuple(zip(*lidx))
447
448        leng = [len(cod) for cod in codec]
449        # mode full : tous False et même longueur
450        fullmode = not max(isfullkeys) and max(leng) == min(leng)
451        # mode default : tous True (idem all(isfullkeys))
452        defmode = min(isfullkeys)
453
454        # not max(isparent) : tous isparent = False
455        if not max(isparent) and (fullmode or defmode):  # mode full ou mode default
456            lindex = [Iindex(idx[2], idx[0], idx[4], idx[1],
457                             reindex=reindex) for idx in lidx]
458            return cls(lindex, reindex=reindex)
459
460        length, crossed = Ilist._init_len_cros(
461            fullmode, leng, isfullkeys, keys, isparent)
462        keyscross = util.canonorder([leng[i] for i in crossed])
463        # name: 0, typevaluedec: 1, codec: 2, parent: 3, keys: 4
464        for ind in range(len(crossed)):
465            lidx[crossed[ind]][4] = keyscross[ind]  # keys
466        for ind in range(len(lidx)):
467            Ilist._init_keys(ind, lidx, length)
468        lindex = [Iindex(idx[2], idx[0], idx[4], idx[1],
469                         reindex=reindex) for idx in lidx]
470        return cls(lindex, reindex=False)

473    def _init_len_cros(fullmode, leng, isfullkeys, keys, isparent):
474        ''' initialization of length and crossed data'''
475        # crossed : pas d'index (isfullindex false), pas de parent(isparent false)
476        crossed = []
477        if fullmode:  # au moins un fullkeys ou une longueur différente
478            length = max(leng)
