117        Iindex constructor.
118
119        *Parameters*

121        - **codec** :  list (default None) - external different values of index (see data model)

125        - **lendefault** : integer (default 0) - default len if no keys is defined

133        if not isinstance(codec, list): codec = [codec]

394    def coupling(self, idx, derived=True):
395        '''
396        Transform indexes in coupled or derived indexes (codec extension).
397        If derived option is True, self.codec is extended and idx codec not,
398        else, both are coupled and both codec are extended.
399
400        *Parameters*
401
402        - **idx** : single Iindex or list of Iindex to be coupled or derived.
403        - **derived** : boolean (default : True)
404
405        *Returns* : tuple with duplicate records (errors)'''
406        if not isinstance(idx, list): index = [idx]
407        else: index = idx
408        idxzip = Iindex.Iext(list(zip(*([self.keys] + [ix.keys for ix in index]))), 
409                             typevalue=None)
410        self.tocoupled(idxzip)
411        if not derived: 
412            for ix in index: ix.tocoupled(idxzip)
413        return self.getduplicates()

415    def couplinginfos(self, other, default=False):
416        '''return a dict with the coupling info between other (lencoupling, rate, 
417        disttomin, disttomax, distmin, distmax, diff, typecoupl)
418
419        *Parameters*
420
421        - **other** : other index to compare
422        - **default** : comparison with default codec 
423
424        *Returns* : dict'''
425        if default: return util.couplinginfos(self.values, other.values)
426        if min(len(self), len(other)) == 0:
427            return {'lencoupling': 0, 'rate': 0, 'disttomin': 0, 'disttomax': 0,
428                    'distmin': 0, 'distmax': 0, 'diff': 0, 'typecoupl': 'null'}
429        ls = len(self.codec)
430        lo = len(other.codec)
431        x0 = max(ls, lo)
432        x1 = ls * lo
433        diff = abs(ls - lo)
434        if min(ls, lo) == 1: 
435            if ls == 1: typec = 'derived'
436            else: typec = 'derive'
437            return {'lencoupling': x0, 'rate': 0, 'disttomin': 0, 'disttomax': 0,
438                    'distmin': x0, 'distmax': x1, 'diff': diff, 'typecoupl': typec}
439        x  = len(util.tocodec([tuple((v1,v2)) for v1, v2 in zip(self.keys, other.keys)]))
440        dic = {'lencoupling': x, 'rate': (x - x0) / (x1 - x0),
441                'disttomin': x - x0,  'disttomax': x1 - x,
442                'distmin': x0, 'distmax': x1, 'diff': diff}
443        if   dic['rate'] == 0 and dic['diff'] == 0: dic['typecoupl'] = 'coupled'
444        elif dic['rate'] == 0 and ls < lo:          dic['typecoupl'] = 'derived'
445        elif dic['rate'] == 0 and ls > lo:          dic['typecoupl'] = 'derive'
446        elif dic['rate'] == 1:                      dic['typecoupl'] = 'crossed'
447        elif ls < lo:                               dic['typecoupl'] = 'linked'
448        else:                                       dic['typecoupl'] = 'link'
449        return dic

495    def getduplicates(self):
496        ''' return tuple of items with duplicate codec'''
497        co = Counter(self.codec)
498        defcodec = list(co - Counter(list(co)))       
499        dkeys  = defaultdict(list)
500        for l,i in zip(self.keys, range(len(self))): dkeys[l].append(i)
501        dcodec = defaultdict(list)
502        for l,i in zip(self.codec, range(len(self.codec))): dcodec[l].append(i)       
503        duplicates = []
504        for item in defcodec: 
505            for codecitem in dcodec[item]: duplicates += dkeys[codecitem]    
506        return tuple(duplicates)

395    def coupling(self, idx, derived=True):
396        '''
397        Transform indexes in coupled or derived indexes (codec extension).
398        If derived option is True, self.codec is extended and idx codec not,
399        else, both are coupled and both codec are extended.
400
401        *Parameters*
402
403        - **idx** : single Iindex or list of Iindex to be coupled or derived.
404        - **derived** : boolean (default : True)
405
406        *Returns* : tuple with duplicate records (errors)'''
407        if not isinstance(idx, list): index = [idx]
408        else: index = idx
409        idxzip = Iindex.Iext(list(zip(*([self.keys] + [ix.keys for ix in index]))), 
410                             typevalue=None)

416    def couplinginfos(self, other, default=False):
417        '''return a dict with the coupling info between other (lencoupling, rate, 
418        disttomin, disttomax, distmin, distmax, diff, typecoupl)
419
420        *Parameters*
421
422        - **other** : other index to compare
423        - **default** : comparison with default codec 
424
425        *Returns* : dict'''
426        if default: return util.couplinginfos(self.values, other.values)
427        if min(len(self), len(other)) == 0:
428            return {'lencoupling': 0, 'rate': 0, 'disttomin': 0, 'disttomax': 0,
429                    'distmin': 0, 'distmax': 0, 'diff': 0, 'typecoupl': 'null'}
430        ls = len(self.codec)
431        lo = len(other.codec)
432        x0 = max(ls, lo)
433        x1 = ls * lo
434        diff = abs(ls - lo)
435        if min(ls, lo) == 1: 
436            if ls == 1: typec = 'derived'
437            else: typec = 'derive'
438            return {'lencoupling': x0, 'rate': 0, 'disttomin': 0, 'disttomax': 0,
439                    'distmin': x0, 'distmax': x1, 'diff': diff, 'typecoupl': typec}
440        x  = len(util.tocodec([tuple((v1,v2)) for v1, v2 in zip(self.keys, other.keys)]))
441        dic = {'lencoupling': x, 'rate': (x - x0) / (x1 - x0),
442                'disttomin': x - x0,  'disttomax': x1 - x,
443                'distmin': x0, 'distmax': x1, 'diff': diff}
444        if   dic['rate'] == 0 and dic['diff'] == 0: dic['typecoupl'] = 'coupled'
445        elif dic['rate'] == 0 and ls < lo:          dic['typecoupl'] = 'derived'
446        elif dic['rate'] == 0 and ls > lo:          dic['typecoupl'] = 'derive'
447        elif dic['rate'] == 1:                      dic['typecoupl'] = 'crossed'
448        elif ls < lo:                               dic['typecoupl'] = 'linked'
449        else:                                       dic['typecoupl'] = 'link'
450        return dic
