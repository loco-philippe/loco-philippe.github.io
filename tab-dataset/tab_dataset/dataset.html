186    def from_csv(cls, filename='dataset.csv', header=True, nrow=None, decode_str=True,
187                 decode_json=True, optcsv={'quoting': csv.QUOTE_NONNUMERIC}):
188        '''
189        Dataset constructor (from a csv file). Each column represents index values.
190
191        *Parameters*
192
193        - **filename** : string (default 'dataset.csv'), name of the file to read
194        - **header** : boolean (default True). If True, the first raw is dedicated to names
195        - **nrow** : integer (default None). Number of row. If None, all the row else nrow
196        - **optcsv** : dict (default : quoting) - see csv.reader options'''
197        if not optcsv:
198            optcsv = {}
199        if not nrow:
200            nrow = -1
201        with open(filename, newline='', encoding="utf-8") as file:
202            reader = csv.reader(file, **optcsv)
203            irow = 0
204            for row in reader:
205                if irow == nrow:
206                    break
207                if irow == 0:
208                    idxval = [[] for i in range(len(row))]
209                    idxname = [''] * len(row)
210                if irow == 0 and header:
211                    idxname = row
212                else:
213                    for i in range(len(row)):
214                        if decode_json:
215                            try:
216                                idxval[i].append(json.loads(row[i]))
217                            except:
218                                idxval[i].append(row[i])
219                        else:
220                            idxval[i].append(row[i])
221                irow += 1
222        lindex = [cls.field_class.from_ntv({name:idx}, decode_str=decode_str) for idx, name in zip(idxval, idxname)]
223        return cls(listidx=lindex, reindex=True)

226    def from_file(cls, filename, forcestring=False, reindex=True, decode_str=False):
227        '''
228        Generate Object from file storage.
229
230         *Parameters*
231
232        - **filename** : string - file name (with path)
233        - **forcestring** : boolean (default False) - if True,
234        forces the UTF-8 data format, else the format is calculated
235        - **reindex** : boolean (default True) - if True, default codec for each Field
236        - **decode_str**: boolean (default False) - if True, string are loaded in json data
237
238        *Returns* : new Object'''
239        with open(filename, 'rb') as file:
240            btype = file.read(1)
241        if btype == bytes('[', 'UTF-8') or btype == bytes('{', 'UTF-8') or forcestring:
242            with open(filename, 'r', newline='', encoding="utf-8") as file:
243                bjson = file.read()
244        else:
245            with open(filename, 'rb') as file:
246                bjson = file.read()
247        return cls.from_ntv(bjson, reindex=reindex, decode_str=decode_str)

249    def merge(self, fillvalue=math.nan, reindex=False, simplename=False):
250        '''
251        Merge method replaces Dataset objects included into its constituents.
252
253        *Parameters*
254
255        - **fillvalue** : object (default nan) - value used for the additional data
256        - **reindex** : boolean (default False) - if True, set default codec after transformation
257        - **simplename** : boolean (default False) - if True, new Field name are
258        the same as merged Field name else it is a composed name.
259
260        *Returns*: merged Dataset '''
261        ilc = copy(self)
262        delname = []
263        row = ilc[0]
264        if not isinstance(row, list):
265            row = [row]
266        merged, oldname, newname = Dataset._mergerecord(self.ext(row, ilc.lname),
267                                                      simplename=simplename)
268        if oldname and not oldname in merged.lname:
269            delname.append(oldname)
270        for ind in range(1, len(ilc)):
271            oldidx = ilc.nindex(oldname)
272            for name in newname:
273                ilc.addindex(self.field(oldidx.codec, name, oldidx.keys))
274            row = ilc[ind]
275            if not isinstance(row, list):
276                row = [row]
277            rec, oldname, newname = Dataset._mergerecord(self.ext(row, ilc.lname),
278                                                       simplename=simplename)
279            if oldname and newname != [oldname]:
280                delname.append(oldname)
281            for name in newname:
282                oldidx = merged.nindex(oldname)
283                fillval = self.field.s_to_i(fillvalue)
284                merged.addindex(
285                    self.field([fillval] * len(merged), name, oldidx.keys))
286            merged += rec
287        for name in set(delname):
288            if name:
289                merged.delindex(name)
290        if reindex:
291            merged.reindex()
292        ilc.lindex = merged.lindex
293        return ilc

295    @classmethod
296    def ext(cls, idxval=None, idxname=None, reindex=True, fast=False):
297        '''
298        Dataset constructor (external index).
299
300        *Parameters*
301
302        - **idxval** : list of Field or list of values (see data model)
303        - **idxname** : list of string (default None) - list of Field name (see data model)'''
304        if idxval is None:
305            idxval = []
306        if not isinstance(idxval, list):
307            return None
308        val = []
309        for idx in idxval:
310            if not isinstance(idx, list):
311                val.append([idx])
312            else:
313                val.append(idx)
314        lenval = [len(idx) for idx in val]
315        if lenval and max(lenval) != min(lenval):
316            raise DatasetError('the length of Iindex are different')
317        length = lenval[0] if lenval else 0
318        idxname = [None] * len(val) if idxname is None else idxname
319        for ind, name in enumerate(idxname):
320            if name is None or name == '$default':
321                idxname[ind] = 'i'+str(ind)
322        lindex = [cls.field_class(codec, name, lendefault=length, reindex=reindex,
323                                  fast=fast) for codec, name in zip(val, idxname)]
324        return cls(lindex, reindex=False)

328    def _mergerecord(rec, mergeidx=True, updateidx=True, simplename=False):
329        #row = rec[0] if isinstance(rec, list) else rec
330        row = rec[0]
331        if not isinstance(row, list):
332            row = [row]
333        var = -1
334        for ind, val in enumerate(row):
335            if val.__class__.__name__ in ['Sdataset', 'Ndataset', 'Observation']:
336                var = ind
337                break
338        if var < 0:
339            return (rec, None, [])
340        ilis = row[var]
341        oldname = rec.lname[var]
342        if ilis.lname == ['i0']:
343            newname = [oldname]
344            ilis.setname(newname)
345        elif not simplename:
346            newname = [oldname + '_' + name for name in ilis.lname]
347            ilis.setname(newname)
348        else:
349            newname = copy(ilis.lname)
350        for name in rec.lname:
351            if name in newname:
352                newname.remove(name)
353            else:
354                updidx = name in ilis.lname and not updateidx
355                ilis.addindex({name: [rec.nindex(name)[0]] * len(ilis)},
356                              merge=mergeidx, update=updidx)
357                #ilis.addindex([name, [rec.nindex(name)[0]] * len(ilis)],
358                #              merge=mergeidx, update=updidx)
359        return (ilis, oldname, newname)
